We consider the system 4aysn.

  Alphabet:

    O : [] --> a 
    cons : [e * f] --> f 
    f : [a * a * a] --> b 
    false : [] --> d 
    filter : [e -> d * f] --> f 
    filter2 : [d * e -> d * e * f] --> f 
    g : [c * c] --> c 
    map : [e -> e * f] --> f 
    nil : [] --> f 
    plus : [a * a] --> a 
    s : [a] --> a 
    true : [] --> d 

  Rules:

    plus(X, O) => X 
    plus(X, s(Y)) => s(plus(X, Y)) 
    f(O, s(O), X) => f(X, plus(X, X), X) 
    g(X, Y) => X 
    g(X, Y) => Y 
    map(F, nil) => nil 
    map(F, cons(X, Y)) => cons(F X, map(F, Y)) 
    filter(F, nil) => nil 
    filter(F, cons(X, Y)) => filter2(F X, F, X, Y) 
    filter2(true, F, X, Y) => cons(X, filter(F, Y)) 
    filter2(false, F, X, Y) => filter(F, Y)
